tgp_southeast_asia_legacy_track = {
	is_shown = {
		has_dlc_feature = all_under_heaven
		OR = {
			government_has_flag = government_is_mandala
			government_has_flag = government_is_wanua
			dynasty = {
				has_dynasty_perk = tgp_sea_legacy_1
			}

			AND = {
				dynasty = {
					dynast = {
						OR = {
							culture = { has_cultural_tradition = tradition_maritime_way_of_life }
							culture = { has_cultural_tradition = tradition_tgp_esoteric_power }
							culture = { has_cultural_tradition = tradition_tgp_barangay_confederations }
						}
					}
				}
			}
		}
	}
}

}
