macro = {
	description = "Checks if the current Oath taken still has its value decided or not"
	definition = "IsOngoingCoronationOath(SetVariable)"
	replace_with = "AddLocalizationIf( EqualTo_CFixedPoint( GetPlayer.MakeScope.Var(SetVariable).GetValue, '(CFixedPoint)0' ), 'coronation_value_still_undecided' )"
}

macro = {
	description = "Checks for cooldown in the title of the decision"
	definition = "CoronationTimer(SetVariable,SetVariable2)"
	replace_with = "AddLocalizationIf( Not( IsGameViewDataShown( 'decision_detail', GetDecisionWithKey(SetVariable) ) ), SetVariable2 )"
}

macro = {
	description = "Checks for remaining variable time of timer in days/months/years"
	definition = "RemainingTime(SetVariable)"
	replace_with = "GetTimeDifferenceWithDays( GetPlayer.MakeScope.VarRemaining(SetVariable) )"
}

macro = {
	description = "Checks for current progress in the activity"
	definition = "CoronationRewardProgress(SetVariable,SetVariable2)"
	replace_with = "AddLocalizationIf( LessThan_CFixedPoint(ActivityWindow.GetSpecialProgressValue, SetVariable), SetVariable2 )"
}
